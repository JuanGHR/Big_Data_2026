from pathlib import Path
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

DATA_DIR = Path("../datasets")
pd.set_option("display.max_columns", 50)

from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans, DBSCAN
from sklearn.metrics import silhouette_score
from sklearn.decomposition import PCA

df = pd.read_csv(DATA_DIR / "clientes_clustering.csv")
scaler = StandardScaler()
X_scaled = scaler.fit_transform(df)

K-Means y silhouette¶

In [2]:
scores = []
for k in range(2, 8):
    km = KMeans(n_clusters=k, n_init="auto", random_state=42)
    labels = km.fit_predict(X_scaled)
    scores.append({"k": k, "silhouette": silhouette_score(X_scaled, labels)})
scores_df = pd.DataFrame(scores)
display(scores_df)

scores_df.plot(x="k", y="silhouette", marker="o", title="Selección de k por silhouette")

Perfilado de clusters¶

In [3]:
k_final = int(scores_df.sort_values("silhouette", ascending=False).iloc[0]["k"])
km = KMeans(n_clusters=k_final, n_init="auto", random_state=42)
df_clustered = df.copy()
df_clustered["cluster"] = km.fit_predict(X_scaled)
display(df_clustered.groupby("cluster").mean().round(2))

# ACTIVIDAD ENTREGABLE - Solución

print("=" * 60)
print("1. MODIFICACIÓN DE HIPERPARÁMETROS")
print("=" * 60)

# Probar con diferentes números de clusters
k_values = range(2, 10)
scores_modified = []

for k in k_values:
    km = KMeans(n_clusters=k, n_init="auto", random_state=42)
    labels = km.fit_predict(X_scaled)
    scores_modified.append({"k": k, "silhouette": silhouette_score(X_scaled, labels)})

scores_df_mod = pd.DataFrame(scores_modified)
print(scores_df_mod)

# Visualizar
scores_df_mod.plot(x="k", y="silhouette", marker="o", title="Selección de k (modificado)")
plt.show()

# Seleccionar mejor k
k_final_mod = int(scores_df_mod.sort_values("silhouette", ascending=False).iloc[0]["k"])
print(f"\n→ Mejor k (silhouette): {k_final_mod}")

print("\n" + "=" * 60)
print("2. INTERPRETACIÓN DE RESULTADOS")
print("=" * 60)

# Modelo final con k seleccionado
km_final = KMeans(n_clusters=k_final_mod, n_init="auto", random_state=42)
df_clustered_final = df.copy()
df_clustered_final["cluster"] = km_final.fit_predict(X_scaled)

print("\nPerfil de clusters:")
perfil = df_clustered_final.groupby("cluster").mean().round(2)
print(perfil)

print("\n→ Interpretación:")
print("- Cada cluster agrupa clientes con comportamientos similares")
print("- Variables como frecuencia, importe_medio, recencia definen los segmentos")
print("- Silhouette score indica qué tan bien separados están los clusters")
print(f"- Con k={k_final_mod}, obtenemos una segmentación balanceada")
